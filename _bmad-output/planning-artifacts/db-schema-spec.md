-- Enum for mission states (assuming 9 states for an AI agent marketplace: created, assigned, in_progress, completed, disputed, resolved, cancelled, paid, refunded)
CREATE TYPE mission_state_enum AS ENUM ('CREATED', 'FUNDED', 'ACCEPTED', 'IN_PROGRESS', 'DELIVERED', 'COMPLETED', 'DISPUTED', 'REFUNDED', 'CANCELLED');

-- Enum for OFAC cache results
CREATE TYPE ofac_result_enum AS ENUM ('CLEAN', 'BLOCKED');

-- Table for providers
-- Stores information about providers in the marketplace, including their wallet, profile, stake, reputation, and status.
CREATE TABLE providers (
    wallet_address TEXT PRIMARY KEY,
    profile TEXT NOT NULL,
    stake_amount DECIMAL NOT NULL,
    reputation_score INTEGER NOT NULL DEFAULT 0,
    status TEXT NOT NULL
);

-- Indexes for providers
CREATE INDEX idx_providers_status ON providers (status);

-- Table for agents
-- Stores AI agents provided by providers, with metadata and status.
CREATE TABLE agents (
    id SERIAL PRIMARY KEY,
    provider_id TEXT NOT NULL REFERENCES providers(wallet_address) ON DELETE CASCADE,
    name TEXT NOT NULL,
    tags TEXT[] NOT NULL,
    metadata_cid TEXT NOT NULL,
    on_chain_id TEXT NOT NULL,
    status TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for agents
CREATE INDEX idx_agents_provider_id ON agents (provider_id);
CREATE INDEX idx_agents_name ON agents (name);
CREATE INDEX idx_agents_status ON agents (status);
CREATE INDEX idx_agents_created_at ON agents (created_at);

-- Table for missions
-- Represents missions or tasks assigned to agents, including client details, state, and financial info.
CREATE TABLE missions (
    id SERIAL PRIMARY KEY,
    client_wallet TEXT NOT NULL,
    agent_id INTEGER NOT NULL REFERENCES agents(id) ON DELETE CASCADE,
    state mission_state_enum NOT NULL,
    amount_usdc DECIMAL NOT NULL,
    fiat_holdback_expires_at TIMESTAMP WITH TIME ZONE,
    dispute_window_expires_at TIMESTAMP WITH TIME ZONE,
    tx_hash TEXT,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for missions
CREATE INDEX idx_missions_client_wallet ON missions (client_wallet);
CREATE INDEX idx_missions_agent_id ON missions (agent_id);
CREATE INDEX idx_missions_state ON missions (state);
CREATE INDEX idx_missions_tx_hash ON missions (tx_hash);
CREATE INDEX idx_missions_created_at ON missions (created_at);

-- Table for mission events (append-only)
-- Logs events related to missions for auditing and indexing purposes.
CREATE TABLE mission_events (
    id SERIAL PRIMARY KEY,
    mission_id INTEGER NOT NULL REFERENCES missions(id) ON DELETE CASCADE,
    event_type TEXT NOT NULL,
    block_number BIGINT NOT NULL,
    tx_hash TEXT NOT NULL,
    log_index INTEGER NOT NULL,
    data JSONB NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (tx_hash, log_index)
);

-- Indexes for mission_events
CREATE INDEX idx_mission_events_mission_id ON mission_events (mission_id);
CREATE INDEX idx_mission_events_event_type ON mission_events (event_type);
CREATE INDEX idx_mission_events_block_number ON mission_events (block_number);
CREATE INDEX idx_mission_events_tx_hash ON mission_events (tx_hash);
CREATE INDEX idx_mission_events_created_at ON mission_events (created_at);

-- Table for reviews
-- Stores reviews of missions, including scores and comments.
CREATE TABLE reviews (
    id SERIAL PRIMARY KEY,
    mission_id INTEGER NOT NULL REFERENCES missions(id) ON DELETE CASCADE,
    reviewer_wallet TEXT NOT NULL,
    scores JSONB NOT NULL,
    comment TEXT,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for reviews
CREATE INDEX idx_reviews_mission_id ON reviews (mission_id);
CREATE INDEX idx_reviews_reviewer_wallet ON reviews (reviewer_wallet);
CREATE INDEX idx_reviews_created_at ON reviews (created_at);

-- Table for indexer state
-- Tracks the state of the blockchain indexer for synchronization.
CREATE TABLE indexer_state (
    chain_id INTEGER PRIMARY KEY,
    last_block_number BIGINT NOT NULL,
    last_block_hash TEXT NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Table for OFAC cache
-- Caches OFAC check results for wallets to avoid repeated queries.
CREATE TABLE ofac_cache (
    wallet_address TEXT PRIMARY KEY,
    result ofac_result_enum NOT NULL,
    checked_at TIMESTAMP WITH TIME ZONE NOT NULL,
    expires_at TIMESTAMP WITH TIME ZONE NOT NULL
);

-- Indexes for ofac_cache
CREATE INDEX idx_ofac_cache_result ON ofac_cache (result);
CREATE INDEX idx_ofac_cache_expires_at ON ofac_cache (expires_at);

-- Table for webhook subscriptions
-- Manages subscriptions for webhooks, including events and status.
CREATE TABLE webhook_subscriptions (
    id SERIAL PRIMARY KEY,
    owner_wallet TEXT NOT NULL,
    url TEXT NOT NULL,
    secret_hash TEXT NOT NULL,
    events TEXT[] NOT NULL,
    active BOOLEAN NOT NULL DEFAULT TRUE
);

-- Indexes for webhook_subscriptions
CREATE INDEX idx_webhook_subscriptions_owner_wallet ON webhook_subscriptions (owner_wallet);
CREATE INDEX idx_webhook_subscriptions_active ON webhook_subscriptions (active);
